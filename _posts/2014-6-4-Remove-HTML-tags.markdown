###### method 1:

```js
function strip(html) {
    var tmp = document.createElement('DIV');
    tmp.innerHTML = html;
    return tmp.textContent || tmp.innerText || '';
}
```

###### use RegEx:

```js
function strip(html) {
    return html.replace(/(<([^>]+)>)/ig, '');
}
```

###### with jQuery:

```js
function strip(html) {
    return $(html).text();
}
```
